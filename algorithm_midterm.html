Sequential Search
Sequential Search 수도코드
SeqSearch(int n, keyType[] S, keyType x) {
    index location = 1;
    while (location <= n && S[location] != x)
        location++;
    if (location > n)
        location = 0;
    return location;
}

BinSearch(int n, keyType[] S, keyType x) {
    index location, low, high, mid;
    low = 1; high = n; location = 0;
    while (low <= high && location == 0) {
        mid = floor((low + high) / 2);
        if (x == S[mid])       location = mid;
        else if (x < S[mid])   high = mid - 1;
        else                   low = mid + 1;
    }
    return location;
}

2.2 Merge Sort
mergeSort
mergeSort(int n, keytype[] S) {
    if (n > 1) {
        const int h = floor(n/2), m = n - h;
        copy S[1:h]       to U[1:h];
        copy S[h+1:n]    to V[1:m];
        mergeSort(h, U);
        mergeSort(m, V);
        merge(h, m, U, V, S);
    }
}

W(n) = n lg n − (n-1) ∈ Θ(n lg n)

2.4 Quick Sort
partition (In-Place)
partition(index low, index high) {
    pivotItem = S[low];
    j = low;
    for (i = low+1; i <= high; i++)
        if (S[i] < pivotItem)
            exchange S[i] and S[++j];
    exchange S[low] and S[j];
    return j;
}

3.2 Floyd's Algorithm
D(k)[i][j] = min( D(k-1)[i][j] , D(k-1)[i][k] + D(k-1)[k][j] )
Floyd
floyd(int n, number W[][], number D[][]) {
    D = W;
    for k=1 to n:
        for i=1 to n:
            for j=1 to n:
                D[i][j] = min(D[i][j], D[i][k] + D[k][j]);
}

T(n) = n³

4.5.4 0/1 Knapsack Problem
🎒 비유
무게 제한 배낭에 이익 최대화하도록 물건 선택.
wi > w: P[i][w] = P[i-1][w]
wi ≤ w: P[i][w] = max( P[i-1][w], pi + P[i-1][w-wi] )

기저: P[0][w]=0, P[i][0]=0  |  답: P[n][W]  |  T = Θ(nW)

과제 #1-1: Three-way MergeSort
mergesort3
mergesort3(int n, keytype[] S) {
    if (n > 1) {
        const int h=floor(n/3), m=floor(n/3), p=n-h-m;
        copy S[1:h]       to U; copy S[h+1:h+m] to V; copy S[h+m+1:n] to W;
        mergesort3(h,U); mergesort3(m,V); mergesort3(p,W);
        merge3(h,m,p,U,V,W,S);
    }
}

과제 #1-2: 동전 거스름돈 경우의 수
j < Si: C[i][j] = C[i-1][j]
j ≥ Si: C[i][j] = C[i-1][j] + C[i][j-Si]
coinChange
coinChange(int N, int S[], int m) {
    for i=0 to m: C[i][0] = 1;
    for j=1 to N: C[0][j] = 0;
    for i=1 to m:
        for j=1 to N:
            C[i][j] = C[i-1][j];
            if (j >= S[i]) C[i][j] += C[i][j - S[i]];
    return C[m][N];
}
T(m, N) = Θ(mN)
